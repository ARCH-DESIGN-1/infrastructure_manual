-- 02_usecase_a.sql
-- Tables for Use Case A: Security Baseline and Configuration Assessment

-- 1. Rules definition table
CREATE TABLE sod_config_rules (
    rule_id        NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    rule_code      VARCHAR2(100)  NOT NULL UNIQUE,
    category       VARCHAR2(100)  NOT NULL, -- PASSWORD_POLICY | MFA | ROLE_MGMT | AUDIT | DATA_SECURITY
    control_name   VARCHAR2(500)  NOT NULL,
    description    VARCHAR2(2000),
    config_path    VARCHAR2(500)  NOT NULL, -- REST API endpoint path
    expected_value VARCHAR2(255)  NOT NULL,
    severity       VARCHAR2(20)   CHECK (severity IN ('CRITICAL','HIGH','MEDIUM','LOW')),
    is_active      VARCHAR2(1)    DEFAULT 'Y' CHECK (is_active IN ('Y','N'))
);

-- 2. Staging configuration settings (extracted from HCM/ERP REST endpoints)
CREATE TABLE sod_config_staging (
    staging_id   NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    run_id       VARCHAR2(100)  NOT NULL REFERENCES sod_assessment_runs(run_id),
    config_path  VARCHAR2(500)  NOT NULL,
    config_value VARCHAR2(4000),
    raw_response CLOB,
    environment  VARCHAR2(20)   NOT NULL,
    extracted_at TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL
);

-- 3. Configuration assessment findings
CREATE TABLE sod_config_findings (
    finding_id     NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    run_id         VARCHAR2(100)  NOT NULL REFERENCES sod_assessment_runs(run_id),
    rule_id        NUMBER         NOT NULL REFERENCES sod_config_rules(rule_id),
    rule_code      VARCHAR2(100),
    config_path    VARCHAR2(500),
    expected_value VARCHAR2(255),
    actual_value   VARCHAR2(4000),
    status         VARCHAR2(10)   CHECK (status IN ('PASS','FAIL')),
    checked_at     TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL
);
