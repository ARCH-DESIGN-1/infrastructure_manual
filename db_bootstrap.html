-- bootstrap.sql
-- Admin-level initialization (run as ADMIN on Autonomous Database)
-- Creates the sod_admin schema, tablespaces, and grants necessary credentials

-- Create tablespace if needed (Note: ADB uses DATA tablespace by default)
-- CREATE TABLESPACE sod_data DATAFILE 'sod_data.dbf' SIZE 100M AUTOEXTEND ON;

-- Create the dedicated schema for all SoD objects
CREATE USER sod_admin IDENTIFIED BY "SoD_Admin_P@ss123!"
  DEFAULT TABLESPACE DATA
  TEMPORARY TABLESPACE TEMP
  QUOTA UNLIMITED ON DATA;

-- Grant required system privileges
GRANT CREATE SESSION, CREATE TABLE, CREATE VIEW, CREATE SEQUENCE,
      CREATE PROCEDURE, CREATE INDEX, CREATE TRIGGER,
      DBMS_CLOUD_EXECUTE, SELECT_CATALOG_ROLE
TO sod_admin;

-- Grant DBMS_CLOUD for OCI Object Storage access from PL/SQL
BEGIN
  DBMS_CLOUD_ADMIN.GRANT_DBMS_CLOUD(username => 'SOD_ADMIN');
END;
