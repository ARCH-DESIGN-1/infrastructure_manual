-- ── Orchestrator: called by DBMS_SCHEDULER at 02:00 UTC
-- ── Assumes connector ran at 01:45 UTC and staged data is ready

CREATE OR REPLACE PROCEDURE prc_run_uca_assessment (
    p_environment   IN VARCHAR2 DEFAULT 'PROD',
    p_trigger       IN VARCHAR2 DEFAULT 'SCHEDULED',
    p_triggered_by  IN VARCHAR2 DEFAULT 'SCHEDULER'
) AS
    v_run_id  VARCHAR2(100) := SYS_GUID();
BEGIN
    -- Step 1: Register the assessment run
    INSERT INTO sod_assessment_runs
      (run_id, use_case, environment, ruleset_version,
       trigger_type, triggered_by, run_mode)
    VALUES
      (v_run_id, 'A', p_environment,
       (SELECT MAX(ruleset_version) FROM sod_config_rules WHERE is_active='Y'),
       p_trigger, p_triggered_by, 'FULL');
    COMMIT;

    -- Step 2: Confirm staging data exists for this environment
    DECLARE
        v_count NUMBER;
    BEGIN
        SELECT COUNT(*) INTO v_count
        FROM   sod_config_staging
        WHERE  environment = p_environment
        AND    extracted_at >= SYSDATE - 1;  -- must be from today's connector run

        IF v_count = 0 THEN
            RAISE_APPLICATION_ERROR(-20001,
                'Staging table is empty for today — connector may not have run. Aborting assessment.');
        END IF;
    END;

    -- Step 3: Execute assessment
    pkg_sod_assess_a.prc_run_assessment(v_run_id, p_environment);

    -- Step 4: Auto-export results to Object Storage
    prc_export_run_results(v_run_id, 'A');

    DBMS_OUTPUT.PUT_LINE('UC-A assessment complete. Run ID: ' || v_run_id);

EXCEPTION WHEN OTHERS THEN
    UPDATE sod_assessment_runs SET status = 'ERROR', completed_at = SYSTIMESTAMP
    WHERE  run_id = v_run_id;
    COMMIT;
    RAISE;
END prc_run_uca_assessment;
/

-- Create the daily assessment job
-- Connector runs at 01:45, assessment at 02:00 — 15 min gap
BEGIN
  DBMS_SCHEDULER.CREATE_JOB(
    job_name        => 'SOD_UCA_DAILY_ASSESSMENT',
    job_type        => 'STORED_PROCEDURE',
    job_action      => 'prc_run_uca_assessment',
    start_date      => SYSTIMESTAMP,
    repeat_interval => 'FREQ=DAILY; BYHOUR=2; BYMINUTE=0',
    end_date        => NULL,   -- runs indefinitely
    enabled         => TRUE,
    auto_drop       => FALSE,
    comments        => 'Use Case A: Security & Controls Configuration Assessment — runs daily at 02:00 UTC after connector (01:45 UTC)'
  );
END;
