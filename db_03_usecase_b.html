-- Tables for Use Case B: Role Hierarchies and Sensitive Access Resolution

-- 1. Sensitive Functions (e.g., Create Payments, Process Payroll)
CREATE TABLE sod_sensitive_functions (
    function_id   NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    function_code VARCHAR2(100)  NOT NULL UNIQUE,
    function_name VARCHAR2(500)  NOT NULL,
    description   VARCHAR2(2000),
    created_at    TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL
);

-- 2. Privileges comprising a sensitive function
CREATE TABLE sod_sensitive_function_privs (
    function_id    NUMBER        NOT NULL REFERENCES sod_sensitive_functions(function_id),
    privilege_name VARCHAR2(255) NOT NULL,
    CONSTRAINT pk_sfp PRIMARY KEY (function_id, privilege_name)
);

-- 3. Entitlements mapping roles to privileges
CREATE TABLE sod_entitlements (
    role_id        NUMBER        NOT NULL REFERENCES sod_roles(role_id),
    privilege_name VARCHAR2(255) NOT NULL,
    granted_at     TIMESTAMP     DEFAULT SYSTIMESTAMP,
    CONSTRAINT pk_ent PRIMARY KEY (role_id, privilege_name)
);

-- 4. Flattened Role Hierarchy snapshot table
CREATE TABLE sod_role_hierarchy_flat (
    ancestor_id NUMBER          NOT NULL REFERENCES sod_roles(role_id),
    role_id     NUMBER          NOT NULL REFERENCES sod_roles(role_id),
    depth       NUMBER          NOT NULL,
    path        VARCHAR2(4000)  NOT NULL,
    CONSTRAINT pk_rhf PRIMARY KEY (ancestor_id, role_id)
);

-- Index flattened lookup columns
CREATE INDEX idx_rhf_ancestor ON sod_role_hierarchy_flat(ancestor_id);
CREATE INDEX idx_rhf_role ON sod_role_hierarchy_flat(role_id);

-- 5. Sensitive Access resolved results
CREATE TABLE sod_sensitive_access_results (
    result_id          NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    run_id             VARCHAR2(100)  NOT NULL REFERENCES sod_assessment_runs(run_id),
    function_id        NUMBER         NOT NULL REFERENCES sod_sensitive_functions(function_id),
    function_code      VARCHAR2(100),
    user_id            NUMBER         NOT NULL REFERENCES sod_users(user_id),
    username           VARCHAR2(255),
    granting_role_code VARCHAR2(200),
    access_path        VARCHAR2(4000),
    path_depth         NUMBER,
    risk_level         VARCHAR2(20),
    is_new             VARCHAR2(1)    DEFAULT 'Y' CHECK (is_new IN ('Y','N')),
    assessed_at        TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL
);
