func.py python 
import io, json, logging, os, uuid, datetime
import oci
from fdk import response

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

BUCKET_NAME = "landing-zone-assessment"

def handler(ctx, data: io.BytesIO = None):
    try:
        file_uuid = str(uuid.uuid4())
        today = datetime.datetime.utcnow().strftime("%Y/%m/%d")
        
        content_type = ctx.Headers().get("content-type", "text/csv")
        ext = "csv"
        if "json" in content_type:
            ext = "json"
        elif "xml" in content_type:
            ext = "xml"
            
        object_name = f"pending/{today}/{file_uuid}.{ext}"

        signer = oci.auth.signers.get_resource_principals_signer()
        osc = oci.object_storage.ObjectStorageClient(config={}, signer=signer)
        namespace = osc.get_namespace().data

        payload_bytes = data.getvalue()
        osc.put_object(
            namespace_name=namespace,
            bucket_name=BUCKET_NAME,
            object_name=object_name,
            put_object_body=payload_bytes,
            content_type=content_type
        )

        logger.info(json.dumps({
            "event": "api_file_received",
            "fileUuid": file_uuid,
            "objectPath": object_name
        }))

        return response.Response(
            ctx,
            status_code=202,
            response_data=json.dumps({
                "status": "Accepted",
                "fileUuid": file_uuid,
                "targetPath": object_name,
                "message": "File upload accepted. Ingestion processing running asynchronously."
            }),
            headers={"Content-Type": "application/json"}
        )
    except Exception as e:
        logger.error(json.dumps({"event": "api_receiver_error", "error": str(e)}))
        return response.Response(ctx, status_code=500, response_data=json.dumps({"error": str(e)}))
