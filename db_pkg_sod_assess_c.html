-- ── Detection: cross-join use-case-B results for both functions of each rule
PROCEDURE prc_run_sod_detection (p_run_id IN VARCHAR2) IS
    v_prev_run_id VARCHAR2(100);
BEGIN
    -- ── Step 1: Resolve sensitive access for this run (Use Case B engine)
    prc_run_sensitive_access(p_run_id);

    -- ── Step 2: Detect conflicts — user holds BOTH function_a AND function_b
    INSERT INTO sod_violations (
        run_id, rule_id, rule_code, user_id, username,
        function_a_code, function_b_code, path_a, path_b, severity, is_new
    )
    SELECT
        p_run_id,
        cr.rule_id,
        cr.rule_code,
        a_access.user_id,
        a_access.username,
        a_access.function_code,
        b_access.function_code,
        a_access.access_path,
        b_access.access_path,
        cr.severity,
        'Y'            -- will be updated in step 3
    FROM
        sod_conflict_rules               cr
        JOIN sod_sensitive_access_results a_access
             ON  a_access.run_id    = p_run_id
             AND a_access.function_id = cr.function_a_id
        JOIN sod_sensitive_access_results b_access
             ON  b_access.run_id    = p_run_id
             AND b_access.function_id = cr.function_b_id
             AND b_access.user_id    = a_access.user_id  -- SAME user holds both
    WHERE  cr.is_active = 'Y';

    -- ── Step 3: Mark violations that existed in prior run as is_new='N'
    UPDATE sod_violations v_new
    SET    v_new.is_new = 'N'
    WHERE  v_new.run_id = p_run_id
    AND EXISTS (
        SELECT 1
        FROM   sod_violations v_prev
        JOIN   sod_assessment_runs ar ON v_prev.run_id = ar.run_id
        WHERE  ar.use_case = 'C' AND ar.status = 'COMPLETE'
        AND    v_prev.user_id = v_new.user_id
        AND    v_prev.rule_id = v_new.rule_id
    );

    -- ── Step 4: Update run summary
    UPDATE sod_assessment_runs
    SET    status = 'COMPLETE', completed_at = SYSTIMESTAMP,
           findings_total = (SELECT COUNT(*) FROM sod_violations WHERE run_id=p_run_id),
           findings_new   = (SELECT COUNT(*) FROM sod_violations WHERE run_id=p_run_id AND is_new='Y')
    WHERE  run_id = p_run_id;
    COMMIT;
END;
/

-- ── Accept a violation with a mitigating control
PROCEDURE prc_accept_violation (
    p_violation_id    IN NUMBER,
    p_exception_type  IN VARCHAR2,
    p_justification   IN VARCHAR2,
    p_control         IN VARCHAR2,
    p_approved_by     IN VARCHAR2,
    p_expiry_date     IN DATE
) IS
    v_exception_id NUMBER;
    v_user_id      NUMBER;
    v_rule_id      NUMBER;
BEGIN
    SELECT user_id, rule_id
    INTO   v_user_id, v_rule_id
    FROM   sod_violations
    WHERE  violation_id = p_violation_id;

    INSERT INTO sod_exceptions (
        rule_id, user_id, exception_type, justification,
        compensating_ctrl, approved_by, approved_at, expiry_date
    ) VALUES (
        v_rule_id, v_user_id, p_exception_type, p_justification,
        p_control, p_approved_by, SYSTIMESTAMP, p_expiry_date
    ) RETURNING exception_id INTO v_exception_id;

    UPDATE sod_violations
    SET    status       = CASE p_exception_type
                             WHEN 'ACCEPTED_RISK'       THEN 'ACCEPTED'
                             WHEN 'MITIGATING_CONTROL'  THEN 'MITIGATED'
                          END,
           exception_id = v_exception_id
    WHERE  violation_id = p_violation_id;
    COMMIT;
END;
/

PROCEDURE prc_export_run_results (
    p_run_id   IN VARCHAR2,
    p_use_case IN VARCHAR2
) IS
    v_export_path VARCHAR2(500);
    v_clob        CLOB;
BEGIN
    v_export_path := 'reports/' || p_use_case || '/'
                  || TO_CHAR(SYSDATE,'YYYY/MM/DD') || '/'
                  || p_run_id || '_results.json';

    -- Build JSON from appropriate view
    SELECT JSON_ARRAYAGG(
        JSON_OBJECT(*) RETURNING CLOB
    ) INTO v_clob
    FROM (
        SELECT * FROM v_violations_with_exceptions
        WHERE run_date >= SYSDATE - 1
    );

    -- Write to Object Storage via Resource Principal
    DBMS_CLOUD.PUT_OBJECT(
        object_uri  => 'https://objectstorage.ap-mumbai-1.oraclecloud.com/n/bmuwto4pxmhp/b/sod-reports/o/'
                    || UTL_URL.ESCAPE(v_export_path, FALSE),
        contents    => UTL_RAW.CAST_TO_RAW(v_clob)
    );
END;
/
END pkg_sod_assess_c;
