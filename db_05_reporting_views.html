-- Analytical and Reporting Views for Dashboards

-- 1. Ingestion health view
CREATE OR REPLACE VIEW v_ingestion_health AS
SELECT
    TRUNC(created_at, 'HH')    AS hour_bucket,
    file_type,
    archive_prefix             AS outcome,
    COUNT(*)                   AS file_count,
    SUM(rows_loaded)           AS total_rows,
    ROUND(AVG(EXTRACT(SECOND FROM (processed_at - created_at))),2) AS avg_proc_sec,
    SUM(CASE WHEN status = 'SUCCESS' THEN 1 ELSE 0 END) AS successful,
    SUM(CASE WHEN status = 'FAILED'  THEN 1 ELSE 0 END) AS failed
FROM   unified_landing_staging
WHERE  created_at >= SYSDATE - 7
GROUP BY TRUNC(created_at,'HH'), file_type, archive_prefix;

-- 2. Violations and Exception Mapping View
CREATE OR REPLACE VIEW v_violations_with_exceptions AS
SELECT
    v.violation_id,
    v.rule_code,
    u.username,
    u.email,
    v.function_a_code,
    v.function_b_code,
    v.severity,
    v.status,
    v.is_new,
    e.exception_type,
    e.approved_by,
    e.expiry_date,
    CASE WHEN e.expiry_date < TRUNC(SYSDATE) THEN 'EXPIRED' ELSE 'VALID' END AS exception_status,
    ar.completed_at                    AS run_date
FROM   sod_violations        v
JOIN   sod_users             u  ON u.user_id    = v.user_id
JOIN   sod_assessment_runs   ar ON ar.run_id    = v.run_id
LEFT JOIN sod_exceptions     e  ON e.exception_id = v.exception_id;

-- 3. Executive KPI Dashboard Summary View
CREATE OR REPLACE VIEW v_assessment_dashboard AS
SELECT
    ar.run_id,
    ar.use_case,
    CASE ar.use_case
        WHEN 'A' THEN 'Security & Controls'
        WHEN 'B' THEN 'Sensitive Access'
        WHEN 'C' THEN 'SoD Violations'
    END                                           AS assessment_name,
    ar.environment,
    ar.ruleset_version,
    ar.trigger_type,
    ar.triggered_by,
    ar.started_at,
    ar.completed_at,
    ROUND(EXTRACT(SECOND FROM (ar.completed_at - ar.started_at)),1) AS duration_sec,
    ar.status,
    ar.findings_total,
    ar.findings_pass,
    ar.findings_fail,
    ar.findings_new,
    ROUND((ar.findings_pass / NULLIF(ar.findings_total, 0)) * 100, 1) AS pass_percentage
FROM   sod_assessment_runs ar;
