prc_flatten_role_hierarchy Stored Procedure

Traverses nested user roles using Oracle recursive CONNECT BY queries to output flat mappings.

prc_flatten_role_hierarchy.sql sql 
-- ── Rebuild the flat hierarchy after every ingestion run
-- ── Uses Oracle CONNECT BY for deep role tree traversal

CREATE OR REPLACE PROCEDURE prc_flatten_role_hierarchy AS
BEGIN
    -- Clear previous snapshot
    DELETE FROM ase_role_hierarchy_flat;

    -- 1. Insert identity mappings (each role inherits itself at depth 0)
    INSERT INTO ase_role_hierarchy_flat (role_id, ancestor_id, depth, path)
    SELECT role_id, role_id, 0, role_common_name
    FROM ase_role_b;

    -- 2. Insert hierarchical relationships (depth > 0)
    INSERT INTO ase_role_hierarchy_flat (role_id, ancestor_id, depth, path)
    SELECT
        child_role_id,
        CONNECT_BY_ROOT parent_role_id AS ancestor_id,
        LEVEL AS depth,
        (SELECT role_common_name FROM ase_role_b WHERE role_id = CONNECT_BY_ROOT parent_role_id) || 
        SYS_CONNECT_BY_PATH((SELECT role_common_name FROM ase_role_b WHERE role_id = child_role_id), ' > ') AS path
    FROM ase_role_role_mbr
    CONNECT BY PRIOR child_role_id = parent_role_id;

    COMMIT;
END prc_flatten_role_hierarchy;
