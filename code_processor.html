import io
import json
import logging
import os
import oci
import oracledb
import base64
from fdk import response

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

QUEUE_OCID = os.environ.get("INGESTION_QUEUE_OCID")
QUEUE_ENDPOINT = os.environ.get("QUEUE_ENDPOINT")
VAULT_SECRET_OCID = os.environ.get("VAULT_SECRET_OCID")
DB_USER = os.environ.get("DB_USER", "sod_admin")
DB_DSN = os.environ.get("DB_DSN")

def get_db_password() -> str:
    signer = oci.auth.signers.get_resource_principals_signer()
    sc     = oci.secrets.SecretsClient(config={}, signer=signer)
    bundle = sc.get_secret_bundle(secret_id=VAULT_SECRET_OCID)
    return base64.b64decode(bundle.data.secret_bundle_content.content).decode("utf-8")

def handler(ctx, data: io.BytesIO = None):
    signer = oci.auth.signers.get_resource_principals_signer()
    qc = oci.queue.QueueClient(config={}, service_endpoint=QUEUE_ENDPOINT, signer=signer)

    # Poll for messages
    msgs = qc.get_messages(queue_id=QUEUE_OCID, visibility_in_seconds=300, limit=1).data.messages
    if not msgs:
        return response.Response(ctx, status_code=200, response_data="No messages in queue")

    msg = msgs[0]
    receipt = msg.receipt
    payload = json.loads(msg.content)
    
    logger.info(f"Processing message: {msg.content}")

    try:
        # Connect to DB
        password = get_db_password()
        conn = oracledb.connect(user=DB_USER, password=password, dsn=DB_DSN)
        cursor = conn.cursor()

        # Setup procedure OUT variables
        status_var = cursor.var(str)
        error_var = cursor.var(str)
        rows_var = cursor.var(int)

        # Execute database parsing and silver merge
        cursor.callproc("pkg_sod_ingest.prc_ingest_file", [
            payload["namespace"],
            payload["bucket"],
            payload["objectName"],
            payload["fileType"],
            status_var,
            error_var,
            rows_var
        ])
        
        status = status_var.getvalue()
        error_msg = error_var.getvalue()
        rows_loaded = rows_var.getvalue()

        conn.close()

        if status == "SUCCESS":
            # ✅ ACK message
            qc.delete_message(queue_id=QUEUE_OCID, message_receipt=receipt)
            logger.info(json.dumps({
                "event": "processing_success",
                "object": payload["objectName"],
                "fileType": payload["fileType"],
                "rows": rows_loaded,
                "deliveryCount": msg.delivery_count
            }))
            return response.Response(ctx, status_code=200, response_data="Success")
        else:
            raise Exception(error_msg)

    except Exception as e:
        # ❌ NACK — do not delete message; visibility will expire
        logger.error(json.dumps({
            "event": "processing_failure",
            "object": payload.get("objectName"),
            "error": str(e),
            "retriesRemaining": 3 - msg.delivery_count
        }))
        return response.Response(ctx, status_code=500, response_data=str(e))

# Build Stage: Pinned dependency container
FROM fnproject/python:3.11-dep AS build-stage
WORKDIR /function
ADD requirements.txt /function/
RUN /python/bin/pip install --target /python/site-packages -r requirements.txt

# Final Stage: Lean execution container
FROM fnproject/python:3.11
WORKDIR /function
COPY --from=build-stage /python /python
ADD . /function/

# Hardened non-root security compliance
RUN groupadd -r fnuser && useradd -r -g fnuser fnuser
RUN chown -R fnuser:fnuser /function /python
USER fnuser

ENTRYPOINT ["/python/bin/fdk", "func.py", "handler"]
