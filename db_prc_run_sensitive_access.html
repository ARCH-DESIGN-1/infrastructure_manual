-- ── Core resolution query: user → function via role hierarchy
CREATE OR REPLACE PROCEDURE prc_run_sensitive_access (p_run_id IN VARCHAR2) AS
BEGIN
    -- Step 1: Ensure hierarchy is fresh
    prc_flatten_role_hierarchy();

    -- Step 2: Resolve user -> sensitive function through full hierarchy and privileges
    INSERT INTO sod_sensitive_access_results (
        run_id, function_id, function_code,
        user_id, username, granting_role_code,
        access_path, path_depth, risk_level, is_new
    )
    SELECT
        p_run_id,
        sf.function_id,
        sf.function_code,
        u.user_id,
        u.username,
        (SELECT role_common_name FROM ase_role_b WHERE role_id = ur.role_id) AS granting_role_code,
        'User:' || u.username || ' > ' || rhf.path || ' > ' || p.privilege_code AS access_path,
        rhf.depth + 1                    AS path_depth,
        sf.risk_level,
        'Y'                              AS is_new
    FROM
        per_users                     u
        JOIN per_user_roles           ur  ON ur.user_id  = u.user_id
        JOIN ase_role_hierarchy_flat  rhf ON rhf.ancestor_id = ur.role_id
        JOIN ase_priv_role_mbr        prm ON prm.role_id = rhf.role_id
        JOIN ase_privilege_b          p   ON p.privilege_id = prm.privilege_id
        JOIN sod_sensitive_function_privs sfp ON sfp.privilege_code = p.privilege_code
        JOIN sod_sensitive_functions  sf  ON sf.function_id = sfp.function_id
    WHERE
        u.active_flag = 'Y'
        AND sf.is_active = 'Y'
        AND ur.active_flag = 'Y'
        AND prm.active_flag = 'Y'
        AND p.active_flag = 'Y';

    -- Step 3: Mark records NOT present in prior run as is_new='N'
    UPDATE sod_sensitive_access_results r_new
    SET    r_new.is_new = 'N'
    WHERE  r_new.run_id = p_run_id
    AND    EXISTS (
        SELECT 1
        FROM   sod_sensitive_access_results r_prev
        JOIN   sod_assessment_runs ar ON r_prev.run_id = ar.run_id
        WHERE  ar.use_case = 'B'
        AND    ar.status = 'COMPLETE'
        AND    r_prev.user_id    = r_new.user_id
        AND    r_prev.function_id = r_new.function_id
    );

    UPDATE sod_assessment_runs
    SET    status = 'COMPLETE', completed_at = SYSTIMESTAMP,
           findings_total = (SELECT COUNT(*) FROM sod_sensitive_access_results WHERE run_id = p_run_id),
           findings_new   = (SELECT COUNT(*) FROM sod_sensitive_access_results WHERE run_id = p_run_id AND is_new='Y')
    WHERE  run_id = p_run_id;
    COMMIT;
END;
