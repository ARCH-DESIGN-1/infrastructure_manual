prc_emit_connector_runs Stored Procedure

Queries scheduled connectors, builds message payloads, and triggers OCI queue messages using REST requests.

-- The scheduler reads the registry and enqueues run messages
-- This runs daily at 02:00 UTC — emits one queue message per active connector

CREATE OR REPLACE PROCEDURE prc_emit_connector_runs AS
    v_run_id        VARCHAR2(100);
    v_payload       CLOB;
    v_now           TIMESTAMP := SYSTIMESTAMP;
BEGIN
    FOR c IN (
        SELECT connector_id, run_mode_default, credential_secret, last_watermark
        FROM   sod_connector_registry
        WHERE  is_active = 'Y'
        AND    schedule_cron IS NOT NULL
    ) LOOP
        v_run_id := SYS_GUID();

        -- Build queue message
        v_payload := JSON_OBJECT(
            'connectorId'   VALUE c.connector_id,
            'runId'         VALUE v_run_id,
            'runMode'       VALUE c.run_mode_default,
            'watermark'     VALUE NVL(c.last_watermark, 'FULL'),
            'triggeredBy'   VALUE 'SCHEDULER',
            'scheduledTime' VALUE TO_CHAR(v_now, 'YYYY-MM-DD"T"HH24:MI:SS"Z"')
        );

        -- Enqueue via DBMS_CLOUD (Resource Principal → OCI Queue)
        DBMS_CLOUD.SEND_REQUEST(
            credential_name => NULL,
            uri             => 'https://cell1.queue.messaging.ap-mumbai-1.oci.oraclecloud.com/20210201/queues/'
                            || '<CONNECTOR_QUEUE_OCID>/messages',
            method          => DBMS_CLOUD.METHOD_POST,
            body            => UTL_RAW.CAST_TO_RAW(JSON_OBJECT('messages' VALUE
                                JSON_ARRAY(JSON_OBJECT('content' VALUE v_payload))))
        );

        -- Audit log
        INSERT INTO sod_connector_run_log (connector_id, run_id, trigger_type, run_mode, status)
        VALUES (c.connector_id, v_run_id, 'SCHEDULED', c.run_mode_default, 'QUEUED');
    END LOOP;
    COMMIT;
END prc_emit_connector_runs;
/

-- Create the scheduler job (runs at 02:00 UTC daily)
BEGIN
    DBMS_SCHEDULER.CREATE_JOB(
        job_name        => 'SOD_CONNECTOR_DAILY_EMIT',
        job_type        => 'STORED_PROCEDURE',
        job_action      => 'prc_emit_connector_runs',
        start_date      => SYSTIMESTAMP,
        repeat_interval => 'FREQ=DAILY; BYHOUR=2; BYMINUTE=0',
        enabled         => TRUE,
        comments        => 'Emits OCI Queue messages for all scheduled connectors'
    );
END;
