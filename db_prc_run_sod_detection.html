-- ── Detection: cross-join use-case-B results for both functions of each rule

CREATE OR REPLACE PROCEDURE prc_run_sod_detection (p_run_id IN VARCHAR2) AS
    v_prev_run_id VARCHAR2(100);
BEGIN
    -- ── Step 1: Resolve sensitive access for this run (Use Case B engine)
    prc_run_sensitive_access(p_run_id);

    -- ── Step 2: Detect conflicts — user holds BOTH function_a AND function_b
    INSERT INTO sod_violations (
        run_id, rule_id, rule_code, user_id, username,
        function_a_code, function_b_code, path_a, path_b, severity, is_new
    )
    SELECT
        p_run_id,
        cr.rule_id,
        cr.rule_code,
        a_access.user_id,
        a_access.username,
        a_access.function_code,
        b_access.function_code,
        a_access.access_path,
        b_access.access_path,
        cr.severity,
        'Y'            -- will be updated in step 3
    FROM
        sod_conflict_rules               cr
        JOIN sod_sensitive_access_results a_access
             ON  a_access.run_id    = p_run_id
             AND a_access.function_id = cr.function_a_id
        JOIN sod_sensitive_access_results b_access
             ON  b_access.run_id    = p_run_id
             AND b_access.function_id = cr.function_b_id
             AND b_access.user_id    = a_access.user_id  -- SAME user holds both
    WHERE  cr.is_active = 'Y';

    -- ── Step 3: Mark violations that existed in prior run as is_new='N'
    UPDATE sod_violations v_new
    SET    v_new.is_new = 'N'
    WHERE  v_new.run_id = p_run_id
    AND EXISTS (
        SELECT 1
        FROM   sod_violations v_prev
        JOIN   sod_assessment_runs ar ON v_prev.run_id = ar.run_id
        WHERE  ar.use_case = 'C' AND ar.status = 'COMPLETE'
        AND    v_prev.user_id = v_new.user_id
        AND    v_prev.rule_id = v_new.rule_id
    );

    -- ── Step 4: Update run summary
    UPDATE sod_assessment_runs
    SET    status = 'COMPLETE', completed_at = SYSTIMESTAMP,
           findings_total = (SELECT COUNT(*) FROM sod_violations WHERE run_id=p_run_id),
           findings_new   = (SELECT COUNT(*) FROM sod_violations WHERE run_id=p_run_id AND is_new='Y')
    WHERE  run_id = p_run_id;
    COMMIT;
END;
