CREATE OR REPLACE PACKAGE BODY pkg_sod_ingest AS

    FUNCTION build_uri (p_namespace IN VARCHAR2, p_bucket IN VARCHAR2, p_object_name IN VARCHAR2) RETURN VARCHAR2 IS
    BEGIN
        RETURN 'https://objectstorage.ap-mumbai-1.oraclecloud.com/n/' 
            || p_namespace || '/b/' || p_bucket || '/o/' || p_object_name;
    END build_uri;

    FUNCTION blob_to_clob (p_blob IN BLOB) RETURN CLOB IS
        v_clob CLOB;
        v_varchar VARCHAR2(32767);
        v_start PLS_INTEGER := 1;
        v_buffer PLS_INTEGER := 32767;
    BEGIN
        DBMS_LOB.CREATETEMPORARY(v_clob, TRUE);
        FOR i IN 1..CEIL(DBMS_LOB.GETLENGTH(p_blob) / v_buffer) LOOP
            v_varchar := UTL_RAW.CAST_TO_VARCHAR2(DBMS_LOB.SUBSTR(p_blob, v_buffer, v_start));
            DBMS_LOB.WRITEAPPEND(v_clob, LENGTH(v_varchar), v_varchar);
            v_start := v_start + v_buffer;
        END LOOP;
        RETURN v_clob;
    END blob_to_clob;

    PROCEDURE prc_validate_file (
        p_namespace IN VARCHAR2, p_bucket IN VARCHAR2,
        p_object_name IN VARCHAR2, p_file_type IN VARCHAR2,
        p_file_size OUT NUMBER
    ) IS
    BEGIN
        IF p_file_type NOT IN ('CSV', 'JSON', 'XML') THEN
            RAISE_APPLICATION_ERROR(-20001, 'UNSUPPORTED_FILE_TYPE');
        END IF;

        DECLARE
            v_metadata CLOB;
        BEGIN
            v_metadata := DBMS_CLOUD.GET_OBJECT_METADATA(
                credential_name => 'OCI_RESOURCE_PRINCIPAL',
                object_uri      => build_uri(p_namespace, p_bucket, p_object_name)
            );
            p_file_size := JSON_VALUE(v_metadata, '$.contentLength');
        EXCEPTION WHEN OTHERS THEN
            RAISE_APPLICATION_ERROR(-20002, 'VALIDATION_FAILED');
        END;
    END prc_validate_file;

    PROCEDURE prc_move_object_prefix (
        p_namespace IN VARCHAR2, p_bucket IN VARCHAR2,
        p_object_name IN VARCHAR2, p_new_prefix IN VARCHAR2
    ) IS
        v_base_name VARCHAR2(500);
        v_target_uri VARCHAR2(1000);
    BEGIN
        v_base_name := SUBSTR(p_object_name, INSTR(p_object_name, '/', -1) + 1);
        v_target_uri := build_uri(p_namespace, p_bucket, p_new_prefix || '/' || TO_CHAR(SYSDATE, 'YYYY/MM/DD') || '/' || v_base_name);
        
        DBMS_CLOUD.COPY_OBJECT(
            credential_name => 'OCI_RESOURCE_PRINCIPAL',
            source_uri      => build_uri(p_namespace, p_bucket, p_object_name),
            target_uri      => v_target_uri
        );
        DBMS_CLOUD.DELETE_OBJECT(
            credential_name => 'OCI_RESOURCE_PRINCIPAL',
            object_uri      => build_uri(p_namespace, p_bucket, p_object_name)
        );
    END prc_move_object_prefix;

    PROCEDURE prc_ingest_json (p_clob IN CLOB, p_staging_id IN NUMBER, p_rows_loaded OUT NUMBER) IS
        v_before NUMBER;
        v_after NUMBER;
    BEGIN
        SELECT COUNT(*) INTO v_before FROM sod_users;

        MERGE INTO sod_users tgt
        USING (
            SELECT jt.user_id, jt.username, jt.email
            FROM   JSON_TABLE(p_clob, '$[*]'
                       COLUMNS (
                           user_id  NUMBER PATH '$.user_id',
                           username VARCHAR2(255) PATH '$.username',
                           email    VARCHAR2(255) PATH '$.email'
                       )
                   ) jt
        ) src
        ON (tgt.user_id = src.user_id)
        WHEN MATCHED THEN UPDATE SET
            tgt.username = src.username, tgt.updated_at = SYSTIMESTAMP
        WHEN NOT MATCHED THEN INSERT (user_id, username, email, created_at)
            VALUES (src.user_id, src.username, src.email, SYSTIMESTAMP);

        SELECT COUNT(*) INTO v_after FROM sod_users;
        p_rows_loaded := v_after - v_before;
    END prc_ingest_json;

    PROCEDURE prc_ingest_file (
        p_namespace     IN  VARCHAR2,
        p_bucket        IN  VARCHAR2,
        p_object_name   IN  VARCHAR2,
        p_file_type     IN  VARCHAR2,
        p_status        OUT VARCHAR2,
        p_error_message OUT VARCHAR2,
        p_rows_loaded   OUT NUMBER
    ) IS
        v_file_size NUMBER := 0;
        v_staging_id NUMBER;
        v_blob BLOB;
        v_clob CLOB;
    BEGIN
        INSERT INTO unified_landing_staging (file_name, file_type, status)
        VALUES (p_object_name, p_file_type, 'RUNNING')
        RETURNING staging_id INTO v_staging_id;
        COMMIT;

        prc_validate_file(p_namespace, p_bucket, p_object_name, p_file_type, v_file_size);

        v_blob := DBMS_CLOUD.GET_OBJECT(
            credential_name => 'OCI_RESOURCE_PRINCIPAL',
            object_uri      => build_uri(p_namespace, p_bucket, p_object_name)
        );
        v_clob := blob_to_clob(v_blob);

        prc_ingest_json(v_clob, v_staging_id, p_rows_loaded);

        UPDATE unified_landing_staging
        SET    status = 'SUCCESS', processed_at = SYSTIMESTAMP, file_size_bytes = v_file_size, rows_loaded = p_rows_loaded, archive_prefix = 'processed'
        WHERE  staging_id = v_staging_id;
        
        prc_move_object_prefix(p_namespace, p_bucket, p_object_name, 'processed');
        
        p_status := 'SUCCESS';
        COMMIT;

    EXCEPTION WHEN OTHERS THEN
        ROLLBACK;
        p_status := 'FAILED';
        p_error_message := SQLERRM;
        
        UPDATE unified_landing_staging
        SET    status = 'FAILED', processed_at = SYSTIMESTAMP, error_code = TO_CHAR(SQLCODE), error_detail = SQLERRM, archive_prefix = 'error'
        WHERE  staging_id = v_staging_id;
        COMMIT;

        BEGIN
            prc_move_object_prefix(p_namespace, p_bucket, p_object_name, 'error');
        EXCEPTION WHEN OTHERS THEN NULL; END;
    END prc_ingest_file;

END pkg_sod_ingest;
