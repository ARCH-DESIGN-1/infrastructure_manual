import io
import json
import logging
import os
import oci
import oracledb
import base64
from fdk import response

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

VAULT_SECRET_OCID = os.environ.get("VAULT_SECRET_OCID")
DB_USER = os.environ.get("DB_USER", "sod_admin")
DB_DSN = os.environ.get("DB_DSN")

def get_db_password() -> str:
    signer = oci.auth.signers.get_resource_principals_signer()
    sc     = oci.secrets.SecretsClient(config={}, signer=signer)
    bundle = sc.get_secret_bundle(secret_id=VAULT_SECRET_OCID)
    return base64.b64decode(bundle.data.secret_bundle_content.content).decode("utf-8")

def query_run_log(run_id):
    password = get_db_password()
    conn = oracledb.connect(user=DB_USER, password=password, dsn=DB_DSN)
    cursor = conn.cursor()
    cursor.execute(
        "SELECT connector_id, status, records_loaded, triggered_by, started_at, completed_at, error_message FROM sod_connector_run_log WHERE run_id = :1",
        [run_id]
    )
    row = cursor.fetchone()
    conn.close()
    if not row:
        return None
    return {
        "connector_id": row[0],
        "status": row[1],
        "records_loaded": row[2],
        "triggered_by": row[3],
        "started_at": row[4],
        "completed_at": row[5],
        "error_message": row[6]
    }

def handler(ctx, data=None):
    try:
        # Extract run_id from Request URL /v1/runs/{runId}/status
        url_parts = ctx.RequestURL().split("/")
        run_id = url_parts[-2] if len(url_parts) >= 2 else ""
        
        if not run_id:
            return response.Response(ctx, status_code=400, response_data=json.dumps({"error": "runId path parameter is required"}))

        row = query_run_log(run_id)
        if not row:
            return response.Response(ctx, status_code=404, response_data=json.dumps({"error": "Run not found"}))
            
        return response.Response(ctx, status_code=200,
            response_data=json.dumps({
                "runId":       run_id,
                "connectorId": row["connector_id"],
                "status":      row["status"],
                "rows":        row["records_loaded"],
                "triggeredBy": row["triggered_by"],
                "startedAt":   str(row["started_at"]),
                "completedAt": str(row["completed_at"]),
                "error":       row["error_message"]
            }),
            headers={"Content-Type": "application/json"}
        )
    except Exception as e:
        logger.error(f"Error querying run log status: {str(e)}")
        return response.Response(ctx, status_code=500, response_data=json.dumps({"error": str(e)}))
