import io, json, logging, os, uuid
import oci
from fdk import response
from connector_registry import ConnectorRegistry
from connector_base import ConnectorConfig
from adb_loader import stream_to_adb

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

QUEUE_OCID     = os.environ["CONNECTOR_QUEUE_OCID"]
QUEUE_ENDPOINT = os.environ["QUEUE_ENDPOINT"]

def handler(ctx, data: io.BytesIO = None):
    signer = oci.auth.signers.get_resource_principals_signer()
    qc     = oci.queue.QueueClient(config={}, service_endpoint=QUEUE_ENDPOINT, signer=signer)

    # Long-poll: wait up to 30s for a message
    msgs = qc.get_messages(
        queue_id=QUEUE_OCID,
        visibility_in_seconds=600,  # 10 min to process + ACK
        timeout_in_seconds=30,
        limit=1
    ).data.messages

    if not msgs:
        return response.Response(ctx, status_code=200,
                         response_data=json.dumps({"status": "idle"}))

    msg     = msgs[0]
    receipt = msg.receipt
    payload = json.loads(msg.content)

    connector_id = payload["connectorId"]
    run_id       = payload["runId"]
    run_mode     = payload.get("runMode", "INCREMENTAL")
    triggered_by = payload.get("triggeredBy", "SCHEDULER")

    ConnectorRegistry.auto_discover()   # loads all /connectors/*.py

    config = ConnectorConfig(
        connector_id   = connector_id,
        source_type    = payload.get("sourceType", ""),
        environment    = os.environ.get("ENVIRONMENT", "PROD"),
        run_id         = run_id,
        run_mode       = run_mode,
        credential_ref = payload.get("credentialRef", ""),
        params         = payload.get("params", {}),
        watermark      = payload.get("watermark")
    )

    connector = ConnectorRegistry.get(connector_id, config)

    try:
        connector.validate()
        records_loaded = stream_to_adb(connector.extract())  # streams to ADB staging

        # ✅ ACK — permanently removes message from queue
        qc.delete_message(queue_id=QUEUE_OCID, message_receipt=receipt)

        logger.info(json.dumps({
            "event": "connector_success", "connectorId": connector_id,
            "runId": run_id, "rows": records_loaded,
            "triggeredBy": triggered_by
        }))
        return response.Response(ctx, status_code=200,
                         response_data=json.dumps({"status": "success", "rows": records_loaded}))

    except Exception as e:
        logger.error(json.dumps({
            "event": "connector_failure", "connectorId": connector_id,
            "error": str(e), "retriesRemaining": 3 - msg.delivery_count
        }))
        return response.Response(ctx, status_code=500,
                         response_data=json.dumps({"status": "failed"}))
    finally:
        connector.close()
connector_registry.py python 
import importlib
import os
import pkgutil
import logging

logger = logging.getLogger(__name__)

class ConnectorRegistry:
    _registry = {}

    @classmethod
    def register(cls, name):
        def decorator(subclass):
            cls._registry[name] = subclass
            return subclass
        return decorator

    @classmethod
    def get(cls, name):
        cls._discover()
        if name not in cls._registry:
            raise KeyError(f"Connector '{name}' is not registered.")
        return cls._registry[name]

    @classmethod
    def _discover(cls):
        pkg_dir = os.path.join(os.path.dirname(__file__), "connectors")
        for _, module_name, _ in pkgutil.walk_packages([pkg_dir]):
            importlib.import_module(f"connectors.{module_name}")

import oracledb
import json
import os
import oci
import base64
import logging

logger = logging.getLogger(__name__)

def get_db_password() -> str:
    signer = oci.auth.signers.get_resource_principals_signer()
    sc     = oci.secrets.SecretsClient(config={}, signer=signer)
    bundle = sc.get_secret_bundle(secret_id=os.environ["VAULT_SECRET_OCID"])
    return base64.b64decode(bundle.data.secret_bundle_content.content).decode("utf-8")

def stream_to_adb(records_iterator, connector_id, run_id):
    password = get_db_password()
    conn = oracledb.connect(
        user=os.environ.get("DB_USER", "sod_admin"),
        password=password,
        dsn=os.environ.get("DB_DSN")
    )
    cursor = conn.cursor()
    
    records_loaded = 0
    records_extracted = 0
    
    batch = []
    for record in records_iterator:
        records_extracted += 1
        batch.append([
            run_id,
            record.entity_type,
            record.entity_key,
            json.dumps(record.payload)
        ])
        
        if len(batch) >= 1000:
            cursor.executemany(
                "INSERT INTO unified_landing_staging (file_name, file_type, status, error_detail) VALUES (:1, 'JSON', 'SUCCESS', :2)",
                [[b[2], b[3]] for b in batch]
            )
            records_loaded += len(batch)
            batch = []
            
    if batch:
        cursor.executemany(
            "INSERT INTO unified_landing_staging (file_name, file_type, status, error_detail) VALUES (:1, 'JSON', 'SUCCESS', :2)",
            [[b[2], b[3]] for b in batch]
        )
        records_loaded += len(batch)

    cursor.execute(
        """UPDATE sod_connector_registry
           SET    last_watermark      = :1,
                  last_successful_run = SYSTIMESTAMP
           WHERE  connector_id = :2""",
        [json.dumps({"watermark": "SUCCESS"}), connector_id]
    )
    
    cursor.execute(
        """UPDATE sod_connector_run_log
           SET    status            = 'SUCCESS',
                  records_extracted = :1,
                  records_loaded    = :2,
                  completed_at      = SYSTIMESTAMP
           WHERE  run_id = :3""",
        [records_extracted, records_loaded, run_id]
    )
    
    conn.commit()
    conn.close()
    return records_extracted, records_loaded

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Iterator, Optional
from datetime import datetime

@dataclass
class ConnectorConfig:
    connector_id:   str       # "fusion-hcm-roles"
    source_type:    str       # REST | FILE | OBJECT_STORAGE | DATABASE | EXTRACT
    environment:    str       # PROD | UAT
    run_id:         str       # unique per execution (UUID)
    run_mode:       str       # FULL | INCREMENTAL
    credential_ref: str       # OCI Vault secret OCID
    params:         dict = field(default_factory=dict)
    watermark:      Optional[str] = None  # high-watermark for incremental loads

@dataclass
class ConnectorRecord:
    source_id:    str
    run_id:       str
    entity_type:  str          # USER | ROLE | PRIVILEGE | CONFIG | ASSIGNMENT
    entity_key:   str          # business natural key
    payload:      dict         # raw extracted fields
    extracted_at: datetime
    checksum:     Optional[str] = None

class BaseConnector(ABC):
    def __init__(self, config: ConnectorConfig):
        self.config = config

    def validate(self) -> None:
        pass

    @abstractmethod
    def extract(self) -> Iterator[ConnectorRecord]:
        ...

    def close(self) -> None:
        pass

    def _get_secret(self, secret_ocid: str) -> str:
        import oci, base64
        signer = oci.auth.signers.get_resource_principals_signer()
        sc = oci.secrets.SecretsClient(config={}, signer=signer)
        bundle = sc.get_secret_bundle(secret_id=secret_ocid)
        return base64.b64decode(bundle.data.secret_bundle_content.content).decode()

from connectors.connector_base import BaseConnector, ConnectorRecord
from connector_registry import ConnectorRegistry
import requests, datetime
from typing import Iterator

@ConnectorRegistry.register("workday-hcm")
class WorkdayHCMConnector(BaseConnector):
    def validate(self) -> None:
        creds = json.loads(self._get_secret(self.config.credential_ref))
        resp = requests.head(f"https://{self.config.params['host']}/ccx/ws/v1", 
                             auth=(creds['username'], creds['password']), timeout=10)
        if resp.status_code != 200:
            raise ConnectionError("Failed connectivity validation to Workday API")

    def extract(self) -> Iterator[ConnectorRecord]:
        creds = json.loads(self._get_secret(self.config.credential_ref))
        with requests.get(f"https://{self.config.params['host']}/ccx/ws/v1/roles", 
                          auth=(creds['username'], creds['password']), stream=True) as r:
            for item in r.json().get('data', []):
                yield ConnectorRecord(
                    source_id=self.config.connector_id,
                    run_id=self.config.run_id,
                    entity_type="ROLE",
                    entity_key=item['roleId'],
                    payload=item,
                    extracted_at=datetime.datetime.utcnow()
                )
fusion_security_config.py python 
import json
import logging
import requests
import datetime
import oracledb
import oci
import base64
import os
from typing import Iterator
from connectors.connector_base import BaseConnector, ConnectorRecord, ConnectorConfig
from connector_registry import ConnectorRegistry

@ConnectorRegistry.register("fusion-security-config")
class FusionSecurityConfigConnector(BaseConnector):
    def validate(self) -> None:
        creds = json.loads(self._get_secret(self.config.credential_ref))
        resp = requests.head(
            f"{self.config.params['host']}/hcmRestApi/resources/11.13.18.05",
            auth=(creds['username'], creds['password']), 
            timeout=10
        )
        if resp.status_code not in (200, 401):
            raise ConnectionError("Failed connectivity validation to Oracle Fusion API")

    def _get_active_rules(self) -> list:
        db_user = os.environ.get("DB_USER", "sod_admin")
        dsn = os.environ.get("DB_DSN")
        
        signer = oci.auth.signers.get_resource_principals_signer()
        sc = oci.secrets.SecretsClient(config={}, signer=signer)
        bundle = sc.get_secret_bundle(secret_id=os.environ["VAULT_SECRET_OCID"])
        password = base64.b64decode(bundle.data.secret_bundle_content.content).decode("utf-8")
        
        conn = oracledb.connect(user=db_user, password=password, dsn=dsn)
        cursor = conn.cursor()
        cursor.execute("SELECT rule_code, source_path FROM sod_config_rules WHERE is_active = 'Y'")
        rules = [{"code": r[0], "path": r[1]} for r in cursor.fetchall()]
        conn.close()
        return rules

    def extract(self) -> Iterator[ConnectorRecord]:
        creds = json.loads(self._get_secret(self.config.credential_ref))
        rules = self._get_active_rules()
        
        for rule in rules:
            url = f"{self.config.params['host']}{rule['path']}"
            try:
                resp = requests.get(url, auth=(creds['username'], creds['password']), timeout=15)
                if resp.status_code == 200:
                    payload = resp.json()
                    val = next(iter(payload.values())) if isinstance(payload, dict) else str(payload)
                    
                    yield ConnectorRecord(
                        entity_type="CONFIG",
                        entity_key=rule['code'],
                        payload={"path": rule['path'], "value": val, "raw": payload},
                        extracted_at=datetime.datetime.utcnow()
                    )
            except Exception as e:
                logging.error(f"Failed to fetch config for rule {rule['code']}: {str(e)}")
