CREATE OR REPLACE PACKAGE BODY pkg_sod_assess_a AS

-- ── Main entry point: evaluate all active rules for a run
PROCEDURE prc_run_assessment (
    p_run_id      IN VARCHAR2,
    p_environment IN VARCHAR2 DEFAULT 'PROD'
) AS
    v_pass   NUMBER := 0;
    v_fail   NUMBER := 0;
BEGIN
    FOR rule IN (
        SELECT *
        FROM   sod_config_rules
        WHERE  is_active = 'Y'
        AND   (effective_to IS NULL OR effective_to >= TRUNC(SYSDATE))
        ORDER BY category, rule_code
    ) LOOP
        prc_evaluate_rule(p_run_id, rule.rule_id, p_environment, v_pass, v_fail);
    END LOOP;

    -- Update run summary
    UPDATE sod_assessment_runs
    SET    status = 'COMPLETE', completed_at = SYSTIMESTAMP,
           findings_total = v_pass + v_fail,
           findings_pass  = v_pass, findings_fail = v_fail
    WHERE  run_id = p_run_id;
    COMMIT;
END prc_run_assessment;

-- ── Evaluate one rule: fetch config value, compare, write finding
PROCEDURE prc_evaluate_rule (
    p_run_id  IN VARCHAR2, p_rule_id IN NUMBER,
    p_env     IN VARCHAR2,
    io_pass   IN OUT NUMBER, io_fail IN OUT NUMBER
) AS
    v_rule       sod_config_rules%ROWTYPE;
    v_observed   VARCHAR2(1000);
    v_result     VARCHAR2(10);
    v_detail     VARCHAR2(4000);
    v_evidence   CLOB;
    v_prior      VARCHAR2(10);
BEGIN
    SELECT * INTO v_rule FROM sod_config_rules WHERE rule_id = p_rule_id;

    -- Fetch observed value from pre-loaded config staging table
    BEGIN
        SELECT config_value, raw_response
        INTO   v_observed, v_evidence
        FROM   sod_config_staging
        WHERE  config_path = v_rule.source_path
        AND    environment = p_env
        AND    ROWNUM = 1;
    EXCEPTION WHEN NO_DATA_FOUND THEN
        v_observed := NULL; v_result := 'ERROR';
        v_detail   := 'Config item not found in staging — connector may not have extracted it';
    END;

    -- Apply comparison operator
    IF v_result IS NULL THEN
        v_result := fn_compare(v_observed, v_rule.expected_value, v_rule.expected_operator);
        IF v_result = 'FAIL' THEN
            v_detail := 'Expected: ' || v_rule.expected_operator || ' ' || v_rule.expected_value
                     || ' | Observed: ' || NVL(v_observed, '(null)');
        END IF;
    END IF;

    -- Get prior run result for run-over-run comparison
    BEGIN
        SELECT result INTO v_prior
        FROM   sod_config_findings f
        JOIN   sod_assessment_runs r ON f.run_id = r.run_id
        WHERE  f.rule_id = p_rule_id
        AND    r.use_case = 'A'
        AND    r.status = 'COMPLETE'
        ORDER BY r.completed_at DESC FETCH FIRST 1 ROWS ONLY;
    EXCEPTION WHEN NO_DATA_FOUND THEN
        v_prior := NULL;
    END;

    -- Write finding
    INSERT INTO sod_config_findings (
        run_id, rule_id, rule_code, result,
        observed_value, expected_value, deviation_detail,
        evidence_json, severity, is_new_finding, prior_result
    ) VALUES (
        p_run_id, p_rule_id, v_rule.rule_code, v_result,
        v_observed, v_rule.expected_value, v_detail,
        v_evidence, v_rule.severity,
        CASE WHEN v_prior IS NULL OR v_prior != v_result THEN 'Y' ELSE 'N' END,
        v_prior
    );

    IF v_result = 'PASS' THEN io_pass := io_pass + 1;
    ELSE                        io_fail := io_fail + 1;
    END IF;
END prc_evaluate_rule;

-- ── Comparison operator: applies EQ | NEQ | GTE | LTE | CONTAINS | REGEX
FUNCTION fn_compare(
    p_observed IN VARCHAR2, p_expected IN VARCHAR2, p_op IN VARCHAR2
) RETURN VARCHAR2 AS
BEGIN
    RETURN CASE p_op
        WHEN 'EQ'       THEN CASE WHEN p_observed = p_expected        THEN 'PASS' ELSE 'FAIL' END
        WHEN 'NEQ'      THEN CASE WHEN p_observed != p_expected       THEN 'PASS' ELSE 'FAIL' END
        WHEN 'GTE'      THEN CASE WHEN TO_NUMBER(p_observed) >= TO_NUMBER(p_expected) THEN 'PASS' ELSE 'FAIL' END
        WHEN 'LTE'      THEN CASE WHEN TO_NUMBER(p_observed) <= TO_NUMBER(p_expected) THEN 'PASS' ELSE 'FAIL' END
        WHEN 'CONTAINS'  THEN CASE WHEN INSTR(p_observed, p_expected) > 0 THEN 'PASS' ELSE 'FAIL' END
        WHEN 'REGEX'    THEN CASE WHEN REGEXP_LIKE(p_observed, p_expected) THEN 'PASS' ELSE 'FAIL' END
        ELSE 'ERROR'
    END;
END fn_compare;

END pkg_sod_assess_a;
