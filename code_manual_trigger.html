import io, json, logging, os, uuid
import oci, oracledb
from fdk import response

logger = logging.getLogger(__name__)

QUEUE_OCID     = os.environ["CONNECTOR_QUEUE_OCID"]
QUEUE_ENDPOINT = os.environ["QUEUE_ENDPOINT"]

def handler(ctx, data: io.BytesIO = None):
    try:
        body         = json.loads(data.getvalue())
        connector_id = body.get("connectorId", "").strip()
        run_mode     = body.get("runMode", "FULL")
        triggered_by = body.get("triggeredBy", "MANUAL_API")

        # ── Validation ──────────────────────────────────────
        if not connector_id:
            return _error(ctx, 400, "connectorId is required")

        if run_mode not in ("FULL", "INCREMENTAL"):
            return _error(ctx, 400, "runMode must be FULL or INCREMENTAL")

        # ── Verify connector registered and active ─────────
        registry_row = lookup_connector(connector_id)
        if not registry_row:
            return _error(ctx, 404,
                f"Connector '{connector_id}' not found or inactive")

        # ── Enqueue run message ────────────────────────────
        run_id = str(uuid.uuid4())
        payload = json.dumps({
            "connectorId":   connector_id,
            "runId":         run_id,
            "runMode":       run_mode,
            "triggeredBy":   triggered_by,
            "credentialRef": registry_row["credential_secret"],
            "params":        json.loads(registry_row["config_json"] or "{}"),
            "watermark":     registry_row["last_watermark"]
                             if run_mode == "INCREMENTAL" else None
        })

        signer = oci.auth.signers.get_resource_principals_signer()
        qc = oci.queue.QueueClient(config={}, service_endpoint=QUEUE_ENDPOINT, signer=signer)
        qc.put_messages(
            queue_id=QUEUE_OCID,
            put_messages_details=oci.queue.models.PutMessagesDetails(
                messages=[oci.queue.models.PutMessagesDetailsEntry(content=payload)]
            )
        )

        # ── Audit log (write to ADB) ───────────────────────
        log_run_start(connector_id, run_id, "MANUAL", run_mode, triggered_by)

        logger.info(json.dumps({
            "event": "manual_trigger_accepted",
            "connectorId": connector_id,
            "runId": run_id, "triggeredBy": triggered_by
        }))

        # ── 202 Accepted — return run_id for polling ───────
        return response.Response(ctx, status_code=202,
            response_data=json.dumps({
                "status":      "accepted",
                "runId":       run_id,
                "connectorId": connector_id,
                "runMode":     run_mode,
                "statusUrl":   f"/sod/v1/runs/{run_id}/status"
            }))

    except Exception as e:
        logger.error(json.dumps({"event": "manual_trigger_error", "error": str(e)}))
        return _error(ctx, 500, "Internal error — check function logs")

def _error(ctx, code, msg):
    return response.Response(ctx, status_code=code,
        response_data=json.dumps({"error": msg, "code": code}))

# Build Stage: Pinned dependency container
FROM fnproject/python:3.11-dep AS build-stage
WORKDIR /function
ADD requirements.txt /function/
RUN /python/bin/pip install --target /python/site-packages -r requirements.txt

# Final Stage: Lean execution container
FROM fnproject/python:3.11
WORKDIR /function
COPY --from=build-stage /python /python
ADD . /function/

# Hardened non-root security compliance
RUN groupadd -r fnuser && useradd -r -g fnuser fnuser
RUN chown -R fnuser:fnuser /function /python
USER fnuser

ENTRYPOINT ["/python/bin/fdk", "func.py", "handler"]
