prc_export_run_results Stored Procedure

Aggregates latest findings into JSON arrays and uploads the reports directly to OCI Object Storage.

prc_export_run_results.sql sql 
CREATE OR REPLACE PROCEDURE prc_export_run_results (
    p_run_id   IN VARCHAR2,
    p_use_case IN VARCHAR2
) AS
    v_export_path VARCHAR2(500);
    v_clob        CLOB;
BEGIN
    v_export_path := 'reports/' || p_use_case || '/'
                  || TO_CHAR(SYSDATE,'YYYY/MM/DD') || '/'
                  || p_run_id || '_results.json';

    -- Build JSON from appropriate view
    SELECT JSON_ARRAYAGG(
        JSON_OBJECT(*) RETURNING CLOB
    ) INTO v_clob
    FROM (
        SELECT * FROM v_violations_with_exceptions
        WHERE run_date >= SYSDATE - 1   -- latest run
    );

    -- Write to Object Storage via Resource Principal
    DBMS_CLOUD.PUT_OBJECT(
        object_uri  => 'https://objectstorage.ap-mumbai-1.oraclecloud.com/n/bmuwto4pxmhp/b/sod-reports/o/'
                    || UTL_URL.ESCAPE(v_export_path, FALSE),
        contents    => UTL_RAW.CAST_TO_RAW(v_clob)
    );
END;
