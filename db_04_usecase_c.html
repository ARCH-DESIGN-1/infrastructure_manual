-- Tables for Use Case C: Segregation of Duties (SoD) Conflict Detection & Exemptions

-- 1. SoD Conflict Matrix (Ruleset)
CREATE TABLE sod_conflict_rules (
    rule_id         NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    rule_code       VARCHAR2(100)  NOT NULL UNIQUE,
    rule_name       VARCHAR2(500)  NOT NULL,
    function_a_id   NUMBER         NOT NULL REFERENCES sod_sensitive_functions(function_id),
    function_b_id   NUMBER         NOT NULL REFERENCES sod_sensitive_functions(function_id),
    conflict_reason VARCHAR2(2000),
    severity        VARCHAR2(20)   CHECK (severity IN ('CRITICAL','HIGH','MEDIUM','LOW')),
    ruleset_version VARCHAR2(20)   DEFAULT '1.0',
    is_active       VARCHAR2(1)    DEFAULT 'Y' CHECK (is_active IN ('Y','N'))
);

-- 2. Exceptions / Mitigating Controls Registry (policy bypass)
CREATE TABLE sod_exceptions (
    exception_id      NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    rule_id           NUMBER         NOT NULL REFERENCES sod_conflict_rules(rule_id),
    user_id           NUMBER         NOT NULL REFERENCES sod_users(user_id),
    exception_type    VARCHAR2(20)   NOT NULL CHECK (exception_type IN ('ACCEPTED_RISK','MITIGATING_CONTROL')),
    justification     VARCHAR2(4000),
    compensating_ctrl VARCHAR2(4000),
    approved_by       VARCHAR2(255)  NOT NULL,
    approved_at       TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL,
    expiry_date       DATE           NOT NULL,
    is_active         VARCHAR2(1)    DEFAULT 'Y' CHECK (is_active IN ('Y','N'))
);

-- 3. SoD Violations Log
CREATE TABLE sod_violations (
    violation_id    NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    run_id          VARCHAR2(100)  NOT NULL REFERENCES sod_assessment_runs(run_id),
    rule_id         NUMBER         NOT NULL REFERENCES sod_conflict_rules(rule_id),
    rule_code       VARCHAR2(100),
    user_id         NUMBER         NOT NULL REFERENCES sod_users(user_id),
    username        VARCHAR2(255),
    function_a_code VARCHAR2(100),
    function_b_code VARCHAR2(100),
    path_a          VARCHAR2(4000),
    path_b          VARCHAR2(4000),
    severity        VARCHAR2(20),
    status          VARCHAR2(20)   DEFAULT 'OPEN' CHECK (status IN ('OPEN','ACCEPTED','MITIGATED','RESOLVED')),
    is_new          VARCHAR2(1)    DEFAULT 'Y' CHECK (is_new IN ('Y','N')),
    exception_id    NUMBER         REFERENCES sod_exceptions(exception_id),
    detected_at     TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL
);
