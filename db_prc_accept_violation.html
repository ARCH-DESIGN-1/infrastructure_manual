-- ── Accept a violation with a mitigating control

CREATE OR REPLACE PROCEDURE prc_accept_violation (
    p_violation_id    IN NUMBER,
    p_exception_type  IN VARCHAR2,
    p_justification   IN VARCHAR2,
    p_control         IN VARCHAR2,
    p_approved_by     IN VARCHAR2,
    p_expiry_date     IN DATE
) AS
    v_exception_id NUMBER;
    v_user_id      NUMBER;
    v_rule_id      NUMBER;
BEGIN
    SELECT user_id, rule_id
    INTO   v_user_id, v_rule_id
    FROM   sod_violations
    WHERE  violation_id = p_violation_id;

    INSERT INTO sod_exceptions (
        rule_id, user_id, exception_type, justification,
        compensating_ctrl, approved_by, approved_at, expiry_date
    ) VALUES (
        v_rule_id, v_user_id, p_exception_type, p_justification,
        p_control, p_approved_by, SYSTIMESTAMP, p_expiry_date
    ) RETURNING exception_id INTO v_exception_id;

    UPDATE sod_violations
    SET    status       = CASE p_exception_type
                             WHEN 'ACCEPTED_RISK'       THEN 'ACCEPTED'
                             WHEN 'MITIGATING_CONTROL'  THEN 'MITIGATED'
                          END,
           exception_id = v_exception_id
    WHERE  violation_id = p_violation_id;
    COMMIT;
END;
