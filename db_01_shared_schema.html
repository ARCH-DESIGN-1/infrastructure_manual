-- 01_shared_schema.sql
-- Creates common ingestion staging, registry, user masters, and reconciliation tables

-- A. Ingestion staging table (Bronze Tier - Immutable ledger log)
CREATE TABLE unified_landing_staging (
    staging_id      NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    file_name       VARCHAR2(500)  NOT NULL,
    file_type       VARCHAR2(10)   NOT NULL CHECK (file_type IN ('CSV','JSON','XML')),
    status          VARCHAR2(20)   DEFAULT 'PENDING' CHECK (status IN ('PENDING','RUNNING','SUCCESS','FAILED')),
    created_at      TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL,
    processed_at    TIMESTAMP,
    delivery_count  NUMBER(3)      DEFAULT 0 NOT NULL,
    max_retries     NUMBER(3)      DEFAULT 3 NOT NULL,
    error_code      VARCHAR2(30),
    error_detail    VARCHAR2(4000),
    queue_receipt   VARCHAR2(512),
    file_size_bytes NUMBER(15),
    rows_loaded     NUMBER(10),
    archive_prefix  VARCHAR2(10)   DEFAULT 'pending' 
                    CONSTRAINT chk_archive_prefix CHECK (archive_prefix IN ('pending','processed','error'))
);

-- B. Core Person Record (Employees/Contractors)
CREATE TABLE per_all_people_f (
    person_id            NUMBER(18)    NOT NULL PRIMARY KEY,
    person_number        VARCHAR2(30)  NOT NULL,
    effective_start_date DATE          NOT NULL,
    effective_end_date   DATE          NOT NULL
);

-- C. User Accounts Table (Silver Tier - User Accounts)
CREATE TABLE per_users (
    user_id            NUMBER(18)    NOT NULL PRIMARY KEY,
    username           VARCHAR2(100) NOT NULL UNIQUE,
    email              VARCHAR2(255) NOT NULL,
    person_id          NUMBER(18)    REFERENCES per_all_people_f(person_id),
    user_guid          VARCHAR2(64),
    active_flag        VARCHAR2(1)   DEFAULT 'Y' CHECK (active_flag IN ('Y','N')),
    suspended_flag     VARCHAR2(1)   DEFAULT 'N' CHECK (suspended_flag IN ('Y','N')),
    start_date         DATE          NOT NULL,
    end_date           DATE
);

-- D. Email Addresses Table
CREATE TABLE per_email_addresses (
    email_address_id   NUMBER(18)    NOT NULL PRIMARY KEY,
    person_id          NUMBER(18)    NOT NULL REFERENCES per_all_people_f(person_id),
    email_address      VARCHAR2(240) NOT NULL,
    primary_flag       VARCHAR2(1)   DEFAULT 'Y' CHECK (primary_flag IN ('Y','N'))
);

-- E. Role Master Definitions
CREATE TABLE ase_role_b (
    role_id            NUMBER(18)    NOT NULL PRIMARY KEY,
    role_common_name   VARCHAR2(240) NOT NULL UNIQUE,
    role_type_code     VARCHAR2(30)  NOT NULL,
    active_flag        VARCHAR2(1)   DEFAULT 'Y' CHECK (active_flag IN ('Y','N'))
);

CREATE TABLE ase_role_tl (
    role_id            NUMBER(18)    NOT NULL REFERENCES ase_role_b(role_id),
    language           VARCHAR2(4)   NOT NULL,
    role_name          VARCHAR2(240) NOT NULL,
    description        VARCHAR2(2000),
    CONSTRAINT pk_ase_role_tl PRIMARY KEY (role_id, language)
);

-- F. User-Role Assignments
CREATE TABLE per_user_roles (
    user_role_id       NUMBER(18)    NOT NULL PRIMARY KEY,
    user_id            NUMBER(18)    NOT NULL REFERENCES per_users(user_id),
    role_id            NUMBER(18)    NOT NULL REFERENCES ase_role_b(role_id),
    active_flag        VARCHAR2(1)   DEFAULT 'Y' CHECK (active_flag IN ('Y','N')),
    start_date         DATE          NOT NULL,
    end_date           DATE
);

-- G. Role Hierarchy (Parent Role inherits Child Role)
CREATE TABLE ase_role_role_mbr (
    parent_role_id     NUMBER(18)    NOT NULL REFERENCES ase_role_b(role_id),
    child_role_id      NUMBER(18)    NOT NULL REFERENCES ase_role_b(role_id),
    active_flag        VARCHAR2(1)   DEFAULT 'Y' CHECK (active_flag IN ('Y','N')),
    CONSTRAINT pk_arr PRIMARY KEY (parent_role_id, child_role_id)
);

-- H. Ingestion assessment runs register
CREATE TABLE sod_assessment_runs (
    run_id          VARCHAR2(100)  PRIMARY KEY,
    use_case        VARCHAR2(5)    NOT NULL CHECK (use_case IN ('A','B','C')),
    environment     VARCHAR2(20)   NOT NULL,
    ruleset_version VARCHAR2(20)   DEFAULT '1.0',
    trigger_type    VARCHAR2(20)   DEFAULT 'MANUAL' CHECK (trigger_type IN ('MANUAL','SCHEDULED')),
    triggered_by    VARCHAR2(255)  DEFAULT 'USER',
    started_at      TIMESTAMP      DEFAULT SYSTIMESTAMP NOT NULL,
    completed_at    TIMESTAMP,
    status          VARCHAR2(20)   DEFAULT 'RUNNING' CHECK (status IN ('RUNNING','COMPLETE','ERROR')),
    findings_total  NUMBER         DEFAULT 0,
    findings_pass   NUMBER         DEFAULT 0,
    findings_fail   NUMBER         DEFAULT 0,
    findings_new    NUMBER         DEFAULT 0
);

-- I. Pluggable Connector Registry
CREATE TABLE sod_connector_registry (
    connector_id        VARCHAR2(100) PRIMARY KEY,
    display_name        VARCHAR2(255) NOT NULL,
    source_type         VARCHAR2(30)  NOT NULL CHECK (source_type IN ('REST_API','FLAT_FILE','OBJECT_STORAGE','DATABASE','EXTRACT')),
    connector_class     VARCHAR2(255) NOT NULL,
    credential_secret   VARCHAR2(500),
    config_json         CLOB,
    schedule_cron       VARCHAR2(100),
    run_mode_default    VARCHAR2(20)  DEFAULT 'INCREMENTAL' CHECK (run_mode_default IN ('FULL','INCREMENTAL')),
    last_watermark      VARCHAR2(100),
    last_successful_run TIMESTAMP,
    error_message       VARCHAR2(4000),
    is_active           VARCHAR2(1)   DEFAULT 'Y' CHECK (is_active IN ('Y','N')),
    created_at          TIMESTAMP     DEFAULT SYSTIMESTAMP NOT NULL
);

-- J. Connector Run Log
CREATE TABLE sod_connector_run_log (
    run_log_id        NUMBER         GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    connector_id      VARCHAR2(100)  REFERENCES sod_connector_registry(connector_id),
    run_id            VARCHAR2(100)  NOT NULL UNIQUE,
    trigger_type      VARCHAR2(20)   CHECK (trigger_type IN ('SCHEDULED','MANUAL')),
    run_mode          VARCHAR2(20),
    started_at        TIMESTAMP      DEFAULT SYSTIMESTAMP,
    completed_at      TIMESTAMP,
    status            VARCHAR2(20)   DEFAULT 'QUEUED' CHECK (status IN ('QUEUED','RUNNING','SUCCESS','FAILED','DLQ')),
    records_extracted NUMBER,
    records_loaded    NUMBER,
    error_message     VARCHAR2(4000),
    triggered_by      VARCHAR2(255)
);

-- K. Reconciliation log (Auditing rows in vs out)
CREATE TABLE sod_recon_log (
    recon_id       NUMBER    GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
    staging_id     NUMBER    NOT NULL REFERENCES unified_landing_staging(staging_id),
    file_name      VARCHAR2(500),
    rows_in_file   NUMBER,
    rows_inserted  NUMBER,
    rows_updated   NUMBER,
    rows_duplicate NUMBER,
    recon_passed   VARCHAR2(1) DEFAULT 'N' CHECK (recon_passed IN ('Y','N')),
    recon_note     VARCHAR2(1000),
    created_at     TIMESTAMP DEFAULT SYSTIMESTAMP NOT NULL
);

-- Enable ADB Automatic Indexing on Silver tables
BEGIN
    DBMS_AUTO_INDEX.CONFIGURE('AUTO_INDEX_MODE', 'IMPLEMENT');
END;
