import io
import json
import logging
import os
import oci
from fdk import response

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

QUEUE_OCID = os.environ.get("INGESTION_QUEUE_OCID")
QUEUE_ENDPOINT = os.environ.get("QUEUE_ENDPOINT")

def handler(ctx, data: io.BytesIO = None):
    try:
        # Parse OCI Event JSON
        event = json.loads(data.getvalue())
        logger.info(f"Received event: {json.dumps(event)}")
        
        event_type = event.get("eventType")
        if event_type != "com.oraclecloud.objectstorage.createobject":
            return response.Response(ctx, status_code=400, response_data="Unsupported event type")

        event_data = event.get("data", {})
        object_name = event_data.get("resourceName")
        bucket_name = event_data.get("additionalDetails", {}).get("bucketName")
        namespace = event_data.get("additionalDetails", {}).get("namespace")
        
        # Extract file type
        ext = object_name.split(".")[-1].upper()
        if ext not in ("CSV", "JSON", "XML"):
            logger.warning(f"Unsupported file extension: {ext}")
            return response.Response(ctx, status_code=400, response_data=f"Unsupported extension: {ext}")

        # Enqueue to OCI Queue
        payload = json.dumps({
            "namespace": namespace,
            "bucket": bucket_name,
            "objectName": object_name,
            "fileType": ext
        })

        signer = oci.auth.signers.get_resource_principals_signer()
        qc = oci.queue.QueueClient(config={}, service_endpoint=QUEUE_ENDPOINT, signer=signer)
        qc.put_messages(
            queue_id=QUEUE_OCID,
            put_messages_details=oci.queue.models.PutMessagesDetails(
                messages=[oci.queue.models.PutMessagesDetailsEntry(content=payload)]
            )
        )
        
        logger.info(f"Enqueued object {object_name} to queue")
        return response.Response(ctx, status_code=200, response_data="Event enqueued successfully")
    except Exception as e:
        logger.error(f"Enqueuer error: {str(e)}")
        return response.Response(ctx, status_code=500, response_data=str(e))
